Starlight Language Tutorial

1. Variables

Use let or define to create variables:

let a = 10
define b = "hello"
sldeploy(a, b)
a: 10 b: hello

2. Arrays

Create arrays and use built-in functions:

let arr = [1,2,3,4,5]
let doubled = map(arr, x => x * 2)
sldeploy(doubled)
[2, 4, 6, 8, 10]

4. Loops

let sum = 0
for i in range(1, 6) {
    sum += i
}
sldeploy(sum)
15

5. Functions

define square = x => x * x
let result = square(5)
sldeploy(result)
25

6. Objects

let person = {
    name: "Alice",
    age: 20,
    greet: () => "Hello"
}
sldeploy(person)
{name: "Alice", age: 20, greet: }

7. Built-in functions

let nums = [1,2,3,4,5]
let evens = filter(nums, x => x % 2 == 0)
sldeploy(evens)
[2, 4]

8. Reduce

let nums = [1, 2, 3, 4, 5]
let total = reduce(nums, (acc, x) => acc + x, 0)
sldeploy(total)
15

9. Keys and Values

let obj = {a: 1, b: 2, c: 3}
sldeploy(keys(obj))
sldeploy(values(obj))
["a", "b", "c"] [1, 2, 3]

10. Range

let r1 = range(5)      // 0..4
let r2 = range(2, 6)   // 2..5
let r3 = range(10, 0, -2) // 10, 8, 6, 4, 2
sldeploy(r1)
sldeploy(r2)
sldeploy(r3)
[0, 1, 2, 3, 4] [2, 3, 4, 5] [10, 8, 6, 4, 2]

11. Ask User Input

let name = ask("What is your name?")
sldeploy("Hello, " + name)
// If user types "Alice" // Hello, Alice

12. Num Conversion

sldeploy(num("123"))   // 123
sldeploy(num(true))      // 1
sldeploy(num(false))     // 0
sldeploy(num([1,2,3]))   // 3
123 1 0 3

13. Str Conversion

sldeploy(str(123))    // "123"
sldeploy(str(true))   // "true"
sldeploy(str([1,2,3])) // "1,2,3"
"123" "true" "1,2,3"

14. Sleep / Delay

print("Start")
await sleep(1000)  // wait 1 second
sldeploy("End")
Start (wait 1 second) End

15. Fetch JSON

let data = await get("https://jsonplaceholder.typicode.com/todos/1")
sldeploy(data)
{userId: 1, id: 1, title: "delectus aut autem", completed: false}
